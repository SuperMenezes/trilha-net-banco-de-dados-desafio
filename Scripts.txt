--QUERY #3

from dbo.Filmes f

--QUERY #4

from dbo.Filmes f
where f.Ano = 1997;

--QUERY #5
select *
from dbo.Filmes f
where f.Ano > 2000;

--QUERY #6
select *
from dbo.Filmes f
where f.Duracao > 100 AND
	  f.Duracao < 150
order by f.Duracao ASC;

--QUERY #7

from dbo.Filmes f
group by f.Ano
order by COUNT(*) DESC;
